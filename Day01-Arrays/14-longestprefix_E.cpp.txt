// Problem 14: longest prefix
// Platform: LeetCode
// Difficulty: Easy
//https:https://leetcode.com/problems/longest-common-prefix
// Time Complexity: O(n), Space Complexity: O(n)


class Solution {
public:
    string longestCommonPrefix(vector<string>& strs) {
        string res= "";
        for(int i=0;i<strs[0].size();i++){
            for(auto s:strs){
                if(s[i]!=strs[0][i]){
                    return res;
                }

            }
            res+=strs[0][i];


        }
        return res;
    }
};
